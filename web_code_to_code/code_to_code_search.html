public String encripta(String senha) {
  try {
    MessageDigest dgt = MessageDigest.getInstance("MD5");
    BASE64Encoder encoder64 = new BASE64Encoder();
    return encoder64.encode(dgt.digest());
  } catch (NoSuchAlgorithmException ns) {
    return senha;
  }
}

public static byte[] getBytes(String str) throws Exception {
  java.security.MessageDigest dgt = null;
  dgt = java.security.MessageDigest.getInst("SHA-1");
  dgt.update(str.getBytes());
  return dgt.digest();
}
